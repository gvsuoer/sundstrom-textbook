\begin{align} f_{3\left( {k + 1} \right)} \amp = f_{3k + 3} \notag\\ \amp = f_{3k + 2} + f_{3k + 1}\notag\\ \amp = \left( {f_{3k + 1} + f_{3k} } \right) + f_{3k + 1} \notag\\ \amp = 2f_{3k + 1} + f_{3k}\tag{36} \end{align}
