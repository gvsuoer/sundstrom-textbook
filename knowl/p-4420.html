\begin{equation*} R_8 = \left\{0, 1, 2, 3, 4, 5, 6, 7 \right\}\text{,} \end{equation*}

\begin{equation*} f ( x ) = \left( x^2 + 2 \right) \pmod 8\text{.} \end{equation*}
